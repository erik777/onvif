org.onvif.ver10.device.wsdl.DeviceService
org.onvif.ver10.events.wsdl.EventService
org.onvif.ver10.media.wsdl.MediaService
org.onvif.ver20.imaging.wsdl.ImagingService
org.onvif.ver20.ptz.wsdl.PtzService
